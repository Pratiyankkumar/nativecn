## Usage Example

<ComponentCode
  title="Basic Breadcrumb Usage with Navigation"
  code={`import { Breadcrumb, BreadcrumbList, BreadcrumbItem, BreadcrumbLink, BreadcrumbPage, BreadcrumbSeparator } from '../components/ui/breadcrumb';
import { useColorScheme, useNavigation } from 'react-native';

export function NavigationBreadcrumb({ path }) {
  const colorScheme = useColorScheme();
  const navigation = useNavigation();
  
  return (
    <Breadcrumb>
      <BreadcrumbList mode={colorScheme}>
        {path.map((item, index) => (
          <React.Fragment key={item.path}>
            <BreadcrumbItem>
              {index === path.length - 1 ? (
                <BreadcrumbPage mode={colorScheme}>{item.label}</BreadcrumbPage>
              ) : (
                <BreadcrumbLink 
                  mode={colorScheme}
                  onPress={() => navigation.navigate(item.path)}
                >
                  {item.label}
                </BreadcrumbLink>
              )}
            </BreadcrumbItem>
            {index < path.length - 1 && (
              <BreadcrumbSeparator mode={colorScheme} />
            )}
          </React.Fragment>
        ))}
      </BreadcrumbList>
    </Breadcrumb>
  );
}`}
/>
