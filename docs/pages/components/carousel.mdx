# Carousel Component

import { Carousel, CarouselContent, CarouselItem, CarouselPrevious, CarouselNext } from '../../components/ui/carousel'
import ComponentPreview, { PreviewModeContext } from '../../components/ComponentPreview'
import ComponentCode from '../../components/ComponentCode'
import { useContext } from 'react'

export const CarouselWithMode = (props) => {
  const mode = useContext(PreviewModeContext) || 'light';
  return <Carousel mode={mode} {...props} />;
}

## Installation

The Carousel component provides a flexible way to showcase content in a scrollable container.

<ComponentCode 
  language="bash"
  code="npx @nativecn/cli add carousel"
  title="Installation Command"
/>

## Usage Examples

<ComponentCode
  title="Image Gallery with Loop"
  code={`import { Carousel, CarouselContent, CarouselItem, CarouselPrevious, CarouselNext } from '@nativecn/ui';
import { View, Image, StyleSheet, useColorScheme } from 'react-native';

export function ImageGallery({ images }) {
  const colorScheme = useColorScheme();
  
  return (
    <Carousel 
      loop 
      mode={colorScheme}
      className="w-full"
    >
      <CarouselContent>
        {images.map((image, index) => (
          <CarouselItem key={index}>
            <Image
              source={{ uri: image.url }}
              style={styles.image}
              resizeMode="cover"
            />
          </CarouselItem>
        ))}
      </CarouselContent>
      <CarouselPrevious variant="default" />
      <CarouselNext variant="default" />
    </Carousel>
  );
}

const styles = StyleSheet.create({
  image: {
    width: '100%',
    height: 400,
    borderRadius: 8,
  },
});`}
/>

## Reference

<ComponentCode
  title="Carousel Props"
  language="typescript"
  code={`interface CarouselProps extends ViewProps {
  // Orientation of the carousel
  orientation?: 'horizontal' | 'vertical';
  
  // Enable infinite loop
  loop?: boolean;
  
  // Number of items (optional, automatically calculated)
  itemCount?: number;
  
  // Light/dark mode
  mode?: 'light' | 'dark';
  
  // Optional className for styling
  className?: string;
}

interface CarouselNavigationProps extends ViewProps {
  // Navigation button variant
  variant?: 'default' | 'outline' | 'ghost';
  
  // Button size
  size?: 'default' | 'sm' | 'lg' | 'icon';
  
  // Optional className for styling
  className?: string;
}`}
/>
